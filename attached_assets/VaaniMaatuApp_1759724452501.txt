package com.sahyadri.vaanimaatu

import android.Manifest
import android.content.pm.PackageManager
import android.os.Bundle
import android.widget.Button
import android.widget.TextView
import android.widget.Toast
import androidx.appcompat.app.AppCompatActivity
import androidx.core.app.ActivityCompat
import androidx.core.content.ContextCompat
import androidx.fragment.app.Fragment
import androidx.room.Room
import com.google.android.material.slider.Slider

class MainActivity : AppCompatActivity() {
    private lateinit var feedbackManager: FeedbackManager
    private lateinit var db: TaskDatabase
    private var points = 0

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        // Initialize database
        db = Room.databaseBuilder(applicationContext, TaskDatabase::class.java, "tasks-db").build()

        // Initialize feedback manager
        feedbackManager = FeedbackManager(this)

        // Request audio permission
        if (ContextCompat.checkSelfPermission(this, Manifest.permission.RECORD_AUDIO) != PackageManager.PERMISSION_GRANTED) {
            ActivityCompat.requestPermissions(this, arrayOf(Manifest.permission.RECORD_AUDIO), 1)
        }

        // UI elements
        val startButton: Button = findViewById(R.id.startButton)
        val stopButton: Button = findViewById(R.id.stopButton)
        val taskSelector: Button = findViewById(R.id.taskSelector)
        val delaySlider: Slider = findViewById(R.id.delaySlider)
        val pointsText: TextView = findViewById(R.id.pointsText)

        // Load points
        points = getSharedPreferences("VaaniMaatuPrefs", MODE_PRIVATE).getInt("points", 0)
        pointsText.text = "Points: $points"

        // Feedback controls
        startButton.setOnClickListener {
            val delayMs = delaySlider.value.toInt()
            feedbackManager.startDAF(delayMs) // Default to DAF; extend for FAF/MAF
            Toast.makeText(this, "Feedback started", Toast.LENGTH_SHORT).show()
        }

        stopButton.setOnClickListener {
            feedbackManager.stop()
            points += 50 // Award points per session
            getSharedPreferences("VaaniMaatuPrefs", MODE_PRIVATE).edit().putInt("points", points).apply()
            pointsText.text = "Points: $points"
            Toast.makeText(this, "Feedback stopped", Toast.LENGTH_SHORT).show()
        }

        // Task selection
        taskSelector.setOnClickListener {
            loadFragment(TaskFragment())
        }
    }

    private fun loadFragment(fragment: Fragment) {
        supportFragmentManager.beginTransaction()
            .replace(R.id.fragmentContainer, fragment)
            .commit()
    }

    override fun onRequestPermissionsResult(requestCode: Int, permissions: Array<out String>, grantResults: IntArray) {
        super.onRequestPermissionsResult(requestCode, permissions, grantResults)
        if (requestCode == 1 && grantResults.isNotEmpty() && grantResults[0] != PackageManager.PERMISSION_GRANTED) {
            Toast.makeText(this, "Audio permission required", Toast.LENGTH_LONG).show()
            finish()
        }
    }
}

package com.sahyadri.vaanimaatu

import android.content.Context
import android.content.pm.PackageManager
import android.media.AudioFormat
import android.media.AudioRecord
import android.media.AudioTrack
import android.media.MediaRecorder
import androidx.core.content.ContextCompat
import java.util.concurrent.LinkedBlockingQueue
import kotlin.math.sin

class FeedbackManager(private val context: Context) {
    private var recorder: AudioRecord? = null
    private var player: AudioTrack? = null
    private val sampleRate = 44100
    private val bufferSize = AudioRecord.getMinBufferSize(sampleRate, AudioFormat.CHANNEL_IN_MONO, AudioFormat.ENCODING_PCM_16BIT)
    private val queue = LinkedBlockingQueue<ByteArray>()

    fun hasPermission(): Boolean = ContextCompat.checkSelfPermission(context, Manifest.permission.RECORD_AUDIO) == PackageManager.PERMISSION_GRANTED

    fun startDAF(delayMs: Int) {
        if (!hasPermission()) return
        recorder = AudioRecord(MediaRecorder.AudioSource.MIC, sampleRate, AudioFormat.CHANNEL_IN_MONO, AudioFormat.ENCODING_PCM_16BIT, bufferSize)
        player = AudioTrack(android.media.AudioManager.STREAM_MUSIC, sampleRate, AudioFormat.CHANNEL_OUT_MONO, AudioFormat.ENCODING_PCM_16BIT, bufferSize, AudioTrack.MODE_STREAM)
        recorder?.startRecording()
        player?.play()
        Thread {
            val buffer = ByteArray(bufferSize)
            val delayFrames = (delayMs * sampleRate / 1000) / bufferSize
            repeat(delayFrames) { queue.offer(ByteArray(bufferSize)) }
            while (true) {
                recorder?.read(buffer, 0, bufferSize)
                queue.offer(buffer.clone())
                player?.write(queue.poll(), 0, bufferSize)
            }
        }.start()
    }

    fun startFAF(pitchShift: Float) { // Simplified pitch shift (placeholder; use FFmpegKit for production)
        if (!hasPermission()) return
        recorder = AudioRecord(MediaRecorder.AudioSource.MIC, sampleRate, AudioFormat.CHANNEL_IN_MONO, AudioFormat.ENCODING_PCM_16BIT, bufferSize)
        player = AudioTrack(android.media.AudioManager.STREAM_MUSIC, sampleRate, AudioFormat.CHANNEL_OUT_MONO, AudioFormat.ENCODING_PCM_16BIT, bufferSize, AudioTrack.MODE_STREAM)
        recorder?.startRecording()
        player?.play()
        Thread {
            val buffer = ByteArray(bufferSize)
            while (true) {
                recorder?.read(buffer, 0, bufferSize)
                // Basic pitch shift: resample buffer (simplified; replace with FFmpegKit)
                player?.write(buffer, 0, bufferSize)
            }
        }.start()
    }

    fun startMAF(noiseLevel: Float) {
        if (!hasPermission()) return
        recorder = AudioRecord(MediaRecorder.AudioSource.MIC, sampleRate, AudioFormat.CHANNEL_IN_MONO, AudioFormat.ENCODING_PCM_16BIT, bufferSize)
        player = AudioTrack(android.media.AudioManager.STREAM_MUSIC, sampleRate, AudioFormat.CHANNEL_OUT_MONO, AudioFormat.ENCODING_PCM_16BIT, bufferSize, AudioTrack.MODE_STREAM)
        recorder?.startRecording()
        player?.play()
        Thread {
            val buffer = ByteArray(bufferSize)
            val noise = generateWhiteNoise(bufferSize, noiseLevel)
            while (true) {
                recorder?.read(buffer, 0, bufferSize)
                for (i in buffer.indices) {
                    buffer[i] = (buffer[i] + noise[i]).toByte() // Mix noise
                }
                player?.write(buffer, 0, bufferSize)
            }
        }.start()
    }

    private fun generateWhiteNoise(size: Int, amplitude: Float): ByteArray {
        val noise = ByteArray(size)
        for (i in 0 until size step 2) {
            val sample = (sin(i.toDouble()) * amplitude * 127).toInt()
            noise[i] = sample.toByte()
            noise[i + 1] = (sample shr 8).toByte()
        }
        return noise
    }

    fun stop() {
        recorder?.stop(); recorder?.release()
        player?.stop(); player?.release()
        queue.clear()
    }
}

# VaaniMaatu
Open-source speech feedback app for Kannada-speaking individuals with stuttering.

## Installation
1. Clone: `git clone https://github.com/sahyadri-team/vaanimaatu`
2. Open in Android Studio.
3. Add NotoSansKannada-Regular.ttf to `app/src/main/assets/fonts/`.
4. Build and run on device/emulator (Min SDK 21).

## Usage
1. Grant microphone permission.
2. Select task (reading, naming, conversation).
3. Adjust delay (50-300ms) and start feedback.
4. Earn points per session; track progress.
